#!/bin/bash

# ======= CẤU HÌNH =======
REMOTE_USER="wasadm"
REMOTE_HOST="vn120158541"
REMOTE_DIR="/appvol/tuan/files-in/"
LOCAL_DEST="/Users/richard/Data/80/18/inwark-file"
PROCESSED_DIR="/Users/richard/Data/80/18/process-file"
OTHER_DIR="$PROCESSED_DIR/other"
NUM_WORKERS=4
RSYNC_INTERVAL=5
HOSTNAME=$(hostname)

# ======= Đẩy file đã xử lý sang server khác =======
DEST_USER="wasadm"
DEST_HOST="vn120158542"
DEST_DIR="/appvol/tuan/files-out/"
PUSH_INTERVAL=5

# ======= LOG =======
LOG_INPUT="/Users/richard/Data/80/18/log/file_sync_input.log"
LOG_PROCESS="/Users/richard/Data/80/18/log/file_sync_process.log"
LOG_OUTPUT="/Users/richard/Data/80/18/log/file_sync_output.log"

mkdir -p "$LOCAL_DEST" "$PROCESSED_DIR" "$OTHER_DIR"
mkdir -p "$(dirname "$LOG_INPUT")"

# ==== Hàm xử lý 1 file ====
process_file() {
    file="$1"
    base_name=$(basename "$file")
    if [[ "$base_name" == ABC-DT*.csv ]]; then
        number=$(echo "$base_name" | grep -oE '[0-9]+')
        new_name="$PROCESSED_DIR/processed-ABC-DT${number}-${HOSTNAME}.csv"
        mv "$file" "$new_name"
        echo "$(date +'%F %T') [WORKER] Đã đổi tên: $base_name -> $(basename "$new_name")" >> "$LOG_PROCESS"
    elif [[ "$base_name" == DT*.fin ]]; then
        number=$(echo "$base_name" | grep -oE '[0-9]+')
        new_name="$PROCESSED_DIR/processed-DT${number}.fin"
        mv "$file" "$new_name"
        echo "$(date +'%F %T') [WORKER] Đã đổi tên: $base_name -> $(basename "$new_name")" >> "$LOG_PROCESS"
    else
        mv "$file" "$OTHER_DIR/"
        echo "$(date +'%F %T') [WORKER] Đã chuyển $base_name vào thư mục other" >> "$LOG_PROCESS"
    fi
}

# ==== Worker pool ====
worker_loop() {
    while true; do
        file=$(find "$LOCAL_DEST" -maxdepth 1 -type f | head -n 1)
        if [ -n "$file" ]; then
            (
                flock -n 200 || exit 1
                [ -f "$file" ] && process_file "$file"
            ) 200>"$file.lock"
            rm -f "$file.lock"
        else
            sleep 2
        fi
    done
}

# ==== Luồng rsync lấy file về ====
rsync_loop() {
    while true; do
        echo "$(date +'%F %T') [RSYNC-IN] Bắt đầu đồng bộ file mới từ $REMOTE_HOST..." >> "$LOG_INPUT"
        # Chạy rsync, chỉ lấy tên file vừa lấy về, log từng file riêng
        rsync -avz --remove-source-files --include="*.csv" --include="*.fin" --exclude="*" \
            "$REMOTE_USER@$REMOTE_HOST:$REMOTE_DIR" "$LOCAL_DEST/" | \
        grep -E "^receiving|^$|^\.\/" | \
        grep -v '^receiving incremental file list' | \
        grep -v '^sent ' | \
        grep -v '^total size is ' | \
        grep -v '^created directory' | \
        grep -v '^$' | \
        while read line; do
            # Dòng bắt đầu ./ là tên file
            fname=$(echo "$line" | sed 's|^\./||')
            if [[ "$fname" != "" && "$fname" != "/" ]]; then
                echo "$(date +'%F %T') [RSYNC-IN] Đã lấy: $fname" >> "$LOG_INPUT"
            fi
        done
        sleep $RSYNC_INTERVAL
    done
}

# ==== Luồng đẩy file đã xử lý qua server khác ====
push_loop() {
    while true; do
        echo "$(date +'%F %T') [RSYNC-OUT] Bắt đầu đẩy file qua $DEST_HOST..." >> "$LOG_OUTPUT"
        # Log các file thực sự đã đẩy thành công và xóa local (remove-source-files)
        rsync -avz --remove-source-files "$PROCESSED_DIR/" "$DEST_USER@$DEST_HOST:$DEST_DIR" | \
        grep -E "^sending|^$|^\.\/" | \
        grep -v '^sending incremental file list' | \
        grep -v '^sent ' | \
        grep -v '^total size is ' | \
        grep -v '^created directory' | \
        grep -v '^$' | \
        while read line; do
            fname=$(echo "$line" | sed 's|^\./||')
            if [[ "$fname" != "" && "$fname" != "/" ]]; then
                echo "$(date +'%F %T') [RSYNC-OUT] Đã đẩy: $fname" >> "$LOG_OUTPUT"
            fi
        done
        sleep $PUSH_INTERVAL
    done
}

# ==== Chạy các worker và các luồng đồng bộ song song ====
for i in $(seq 1 $NUM_WORKERS); do
    worker_loop &
done

rsync_loop &
push_loop &

wait
