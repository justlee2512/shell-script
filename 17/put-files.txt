#!/bin/bash

# ======================= CẤU HÌNH =======================
REMOTE_USER="wasadm"
REMOTE_HOST="vn120158541"
REMOTE_DIR="/appvol/tuan/files-in/"
LOCAL_DEST="/Users/richard/Data/80/18/inwark-file"
PROCESSED_DIR="/Users/richard/Data/80/18/process-file"
OTHER_DIR="$PROCESSED_DIR/other"
NUM_WORKERS=4
RSYNC_INTERVAL=5
HOSTNAME=$(hostname)

# Đẩy file đã xử lý sang server khác
DEST_USER="wasadm"
DEST_HOST="vn120158542"
DEST_DIR="/appvol/tuan/files-out/"
PUSH_INTERVAL=5

# Thư mục log
LOG_DIR="/Users/richard/Data/80/18/log"
mkdir -p "$LOCAL_DEST" "$PROCESSED_DIR" "$OTHER_DIR" "$LOG_DIR"

# ========== HÀM GHI LOG (chia theo ngày & loại log) ==========
log_file() {
    # $1 = loại log: in | process | out
    # $2 = tên file
    local log_type="$1"
    local fname="$2"
    local log_path="$LOG_DIR/$(date +'%F')-${log_type}.log"
    echo "$(date +'%T') $fname" >> "$log_path"
}

# ========== XỬ LÝ FILE ==========
process_file() {
    file="$1"
    base_name=$(basename "$file")
    if [[ "$base_name" == ABC-DT*.csv ]]; then
        number=$(echo "$base_name" | grep -oE '[0-9]+')
        new_name="$PROCESSED_DIR/processed-ABC-DT${number}-${HOSTNAME}.csv"
        mv "$file" "$new_name"
        log_file "process" "$(basename "$new_name")"
    elif [[ "$base_name" == DT*.fin ]]; then
        number=$(echo "$base_name" | grep -oE '[0-9]+')
        new_name="$PROCESSED_DIR/processed-DT${number}.fin"
        mv "$file" "$new_name"
        log_file "process" "$(basename "$new_name")"
    else
        mv "$file" "$OTHER_DIR/"
        log_file "process" "$base_name"
    fi
}

# ========== WORKER POOL ==========
worker_loop() {
    while true; do
        # Chỉ lấy file .csv hoặc .fin (không lấy .lock, file lạ)
        file=$(find "$LOCAL_DEST" -maxdepth 1 -type f \( -name "*.csv" -o -name "*.fin" \) | head -n 1)
        if [ -n "$file" ]; then
            (
                flock -n 200 || exit 1
                [ -f "$file" ] && process_file "$file"
            ) 200>"$file.lock"
            rm -f "$file.lock"
        else
            sleep 2
        fi
    done
}

# ========== RSYNC LẤY FILE VỀ ==========
rsync_loop() {
    while true; do
        rsync -avz --remove-source-files --include="*.csv" --include="*.fin" --exclude="*" \
            "$REMOTE_USER@$REMOTE_HOST:$REMOTE_DIR" "$LOCAL_DEST/" | \
        grep -E "^\.\/" | grep -v '^created directory' | grep -v '^$' | while read line; do
            fname=$(echo "$line" | sed 's|^\./||')
            if [[ "$fname" != "" && "$fname" != "/" ]]; then
                log_file "in" "$fname"
            fi
        done
        sleep $RSYNC_INTERVAL
    done
}

# ========== RSYNC ĐẨY FILE RA SERVER KHÁC ==========
push_loop() {
    while true; do
        rsync -avz --remove-source-files "$PROCESSED_DIR/" "$DEST_USER@$DEST_HOST:$DEST_DIR" | \
        grep -E "^\.\/" | grep -v '^created directory' | grep -v '^$' | while read line; do
            fname=$(echo "$line" | sed 's|^\./||')
            if [[ "$fname" != "" && "$fname" != "/" ]]; then
                log_file "out" "$fname"
            fi
        done
        sleep $PUSH_INTERVAL
    done
}

# ========== KHỞI ĐỘNG WORKER & CÁC LUỒNG ==========
for i in $(seq 1 $NUM_WORKERS); do
    worker_loop &
done

rsync_loop &
push_loop &

wait
